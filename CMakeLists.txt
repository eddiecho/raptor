cmake_minimum_required(VERSION 3.10)

project(Raptor VERSION 0.0.1 LANGUAGES CXX)

set(VERSION_HEADER "// This file is autogenerated, do not edit
#define Raptor_VERSION_MAJOR ${Raptor_VERSION_MAJOR}
#define Raptor_VERSION_MINOR ${Raptor_VERSION_MINOR}
#define Raptor_VERSION_PATCH ${Raptor_VERSION_PATCH}
"
)

file(WRITE ${CMAKE_CURRENT_SOURCE_DIR}/src/raptor_version.h "${VERSION_HEADER}")

set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_STANDARD_REQUIRED True)
set(SRC ./src/)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ./bin)

if (NOT CMAKE_BUILD_TYPE)
    set(CMAKE_BUILD_TYPE Debug)
endif()

set(CMAKE_CXX_FLAGS "-Wall -Wextra")
set(CMAKE_CXX_FLAGS_DEBUG "-O1 -gdwarf -DDEBUG")
set(CMAKE_CXX_FLAGS_RELEASE "-O2")

add_executable(raptor src/raptor.cpp)
set_target_properties(raptor PROPERTIES LINKER_LANGUAGE CXX)

configure_file(src/raptor_version.h src/raptor_version.h)
target_include_directories(raptor PUBLIC "src")
